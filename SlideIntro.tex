\begin{frame}
  \frametitle{部分固有対計算}
  \begin{block}{Def.~固有対 $\left( \lambda_j,\bm{q}_j \right)$}
    \begin{itemize}
    \item 実$n$次\textcolor{red}{対称}行列$A$の固有対$\left( \lambda_j,\bm{q}_j \right)$は，以下を満たす:  
      \begin{center}
	$A \bm{q}_j=\lambda_j \bm{q}_j,\ \quad j=1,\ \dots,\ n$
      \end{center}
      \begin{itemize}
      \item $\lambda_j\ (\in \mathbb{R})$: $A$の固有値 ($\lambda_1 \ge \cdots \ge \lambda_n$)
      \item $\bm{q}_j\ (\in \mathbb{R}^n)$: $\lambda_j$に対応する固有ベクトル
      \end{itemize}
    \end{itemize}
  \end{block}
  
  \begin{itemize}
  \item \textcolor{blue}{部分}固有対計算の応用:
    \begin{itemize}
    \item 振動解析 / 分子軌道法
    \item カーネル主成分分析 [Sch\"{o}lkopf-Smola-M\"{u}ller (1998)]
    \item[\checkmark] 部分特異対計算
    \end{itemize}
  \item[\checkmark] 問題の大規模化 $\to$ \textcolor{red}{並列計算}による高速化
    \begin{itemize}
    \item マルチコア計算（共有メモリ, 1ノード）
    \item 大規模分散並列計算（スーパーコンピュータ）
    \item アクセラレータ（Intel Xeon Phi Co-processor, GPU）
    \end{itemize}
  \end{itemize}
\end{frame}

\begin{frame}
  \frametitle{部分固有対計算の方法}
  \input{one_phase_tridiag_evd_en.tex}
  \begin{itemize}
  \item $A$から直接計算するのではなく，以下の3段階に分けて計算
    \begin{enumerate}
    \item 3重対角化 [Wilkinson (1962), etc.]
      \begin{align*}
        T:= & V_T A V_T^\top
      \end{align*}
    \item 実対称3重対角行列$T$の固有対を計算
    \item 逆変換: $T$の固有ベクトルから$A$の固有ベクトルを計算
      \begin{align*}
        \bm{q}_k := & V_T \bm{q}_k \quad \text{for $k=1,\dots,\ell$}
      \end{align*}
    \end{enumerate}
  \end{itemize}
\end{frame}

  \input{one_phase_tridiag_evd_en.tex}

  \item 1-step tridiagonalization (3重対角化): Full-to-Tri

    \item Householder tridiagonalization [Wilkinson (1962)]
    \item[\checkmark] Dongarra's algorithm [Dongarra et al.~(1989)]: ブロック版
    \end{itemize}
  \end{itemize}

  \input{two_phase_tridiag_evd_en.tex}
  \begin{itemize}
  \item 2-step tridiagonalization: Full-to-Band + Band-to-Tri
    \begin{itemize}
    \item[\checkmark] Bischof's algorithm [Bischof-Lang-Sun (1993/2000)]
      \begin{itemize}
      \item $A \to B$: Dongarra法よりも高性能
      \item[\checkmark] $B \to T$のスケーラビリティが課題
      \end{itemize}
    \item $B \to T$の性能は逆変換の計算量とのトレードオフ
    \end{itemize}
  \end{itemize}
\end{frame}

\begin{frame}
  \frametitle{部分固有対計算の並列実装 (2/2)}
  \input{one_phase_band_evd_en.tex}
  \begin{itemize}
  \item 3重対角行列$T$ではなく，\underline{帯行列$B$に変換}
    \begin{enumerate}
    \item 帯行列化 [Bischof-Lang-Sun (1993/2000)]
      \begin{align*}
        B:= & V_B A V_B^\top
      \end{align*}
    \item 実対称帯行列$B$の固有対を計算
    \item 逆変換: $B$の固有ベクトルから$A$の固有ベクトルを計算
      \begin{align*}
        \bm{q}_k := & V_B \bm{q}_k \quad \text{for $k=1,\dots,\ell$}
      \end{align*}
    \end{enumerate}
  \end{itemize}
\end{frame}

\begin{frame}
  \frametitle{実対称3重対角行列の固有対計算アルゴリズム}
  \begin{block}{実対称3重対角行列の固有対計算アルゴリズム}
    \begin{itemize}
    \item 全固有対計算アルゴリズム \underline{$O(n^3)$}: 
      \begin{itemize}
      \item QR法 [Bowdler-Martin-Reinsch-Wilkinson (1968)]
      \item 分割統治法 [Cuppen (1980) / Gu-Eisenstat (1995)]
      \end{itemize}
    \item 部分固有対計算アルゴリズム: 
      \begin{itemize}
      \item 2分法+逆反復法 [Kahan (1966), etc.] \underline{$O(\ell^2 n)$}
        \begin{itemize}
        \item {\color{red}2分法}: 大粒度に並列化でき, 相対精度の意味で高精度な固有値計算法
        \item {\color{blue}逆反復法}: 再直交化計算のため, 計算量が多く並列化も困難
        \end{itemize}
      \item MRRR法 [Dhillon (1997), etc.] \underline{$O(\ell n)$}
        \begin{itemize}
        \item 2分法+逆反復法の改良アルゴリズム
        \item 特定の行列では, 計算量が増加してしまう
        \item ロバストに高精度な計算を実現できるわけではない
        \end{itemize}
      \end{itemize}
    \end{itemize}
  \end{block}

  \begin{itemize}
  \item[\checkmark] 高速・高精度な部分固有対計算の実現 $\to$ \textcolor{red}{逆反法の高速化}
    \begin{itemize}
    \item Chapter 3 \& 4: $T$向け逆反復法の並列計算機向け高速化
    \end{itemize}
  \end{itemize}
  
  \end{frame}

\begin{frame}
  \frametitle{特異対計算}
  \begin{block}{Def. 特異対 $(\sigma_i,\ \bm{u}_i,\ \bm{v}_i)$}
    \begin{itemize}
    \item 実長方行列$M \in \mathbb{R}^{m \times n}$ ($\operatorname{rank}(M)=r,\ r \le \min(m,n)$)の特異対 $(\sigma_i,\ \bm{u}_i,\ \bm{v}_i)$は以下を満たす組:
      \begin{align*}
        \begin{aligned}
	  M \bm{v}_i & = \sigma_i \bm{u}_i\\
	  M^\top \bm{u}_i & = \sigma_i \bm{v}_i
	\end{aligned}
      	\quad (i=1,\ \dots,\ r)
      \end{align*}
      \begin{itemize}
      \item $\sigma_i \in \mathbb{R}$: $M$の特異値 ($\sigma_1 \ge \dots \ge \sigma_r > 0$)
      \item $\bm{u}_i \in \mathbb{R}^m$: $\sigma_i$に対応する左特異ベクトル
      \item $\bm{v}_i \in \mathbb{R}^n$: $\sigma_i$に対応する右特異ベクトル
      \end{itemize}
    \end{itemize}
  \end{block}

  \begin{itemize}
  \item 固有対との関係:
    \begin{align*}
      \begin{bmatrix}
        & M \\
        M^\top & 
      \end{bmatrix}
      \begin{bmatrix}
        \bm{u}_i\\ \bm{v}_i
      \end{bmatrix}
      =
      \sigma_i
      \begin{bmatrix}
        \bm{u}_i\\ \bm{v}_i
      \end{bmatrix}
    \end{align*}
  \end{itemize}
\end{frame}

\begin{frame}
  \frametitle{大規模疎行列向け部分特異対計算のためのアルゴリズム}
  \begin{block}{Golub-Kahan-Lanczos (GKL) 法 [Golub-Kahan (1965)]}
    \begin{itemize}
    \item Krylov部分空間に基づく反復法
    \item $k$反復目において，$k$次\textcolor{red}{2重対角行列$B_k$}を生成
    \end{itemize}
  \end{block}
  \input{rgkl_en.tex}
  \begin{itemize}
  \item GKL法による大規模疎行列の部分特異対計算: 
    \begin{itemize}
    \item[\checkmark] $k_{\textrm{end}}$ (\textcolor{red}{$k_{\textrm{end}} \ll n$})反復で計算終了 比較的少ない計算コスト
    \end{itemize}
  \end{itemize}
\end{frame}

\begin{frame}
  \frametitle{上2重対角行列$B_k$の部分特異対計算}
  \begin{itemize}
  \item 実対称3重対角行列の固有値問題に帰着: \\
    \input{svd_bidiag_arg_en.tex}
    \begin{itemize}
    \item Transform $B_k$ into $T_{2k}$ [Golub-Kahan (1965)]
      \begin{itemize}
      \item $B_k$の特異値が$T_{2k}$の正固有値に対応
      \item[\checkmark] 行と列の入れ替えのみで実現可能
      \end{itemize}
    \item 拡大行列 $T_{2k}$（$2k$次実対称3重対角行列）の部分固有対計算
      \begin{itemize}
      \item \textcolor{red}{最大のものから$\ell$個}の固有値と対応する固有ベクトルのみが必要
      \item[\checkmark] \underline{2分法+逆反復法}が適用可能
      \end{itemize}
    \end{itemize}
  \end{itemize}
  \begin{itemize}
  \item[\checkmark] 上2重対角行列$B_k$の特異対は, 停止条件にも用いられる
  \end{itemize}
\end{frame}
