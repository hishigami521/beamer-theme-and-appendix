\begin{frame}
  \frametitle{部分固有対計算}
  \begin{block}{Def.~eigenpairs $\left( \lambda_j,\bm{q}_j \right)$}
    \begin{itemize}
    \item $\left( \lambda_j,\bm{q}_j \right)$: eigenpairs (eigenvalues and eigenvectors) of an $n \times n$ \textcolor{red}{real symmetric} matrix $A$ s.t. 
      \begin{center}
	$A \bm{q}_j=\lambda_j \bm{q}_j,\ \quad j=1,\ \dots,\ n$
      \end{center}
      \begin{itemize}
      \item $\lambda_j\ (\in \mathbb{R})$: eigenvalue of $A$ ($\lambda_1 < \cdots < \lambda_n$)
      \item $\bm{q}_j\ (\in \mathbb{R}^n)$: the corresponding eigenvector to $\lambda_j$
      \end{itemize}
    \end{itemize}
  \end{block}
  
  \begin{itemize}
  \item \textcolor{blue}{部分}固有対計算の応用:
    \begin{itemize}
    \item 振動解析 / 分子軌道法
    \item カーネル主成分分析 [Sch\"{o}lkopf-Smola-M\"{u}ller (1998)]
    \item[\checkmark] 部分特異対計算
    \end{itemize}
  \item[\checkmark] 問題の大規模化 $\to$ \textcolor{red}{並列計算}による高速化
    \begin{itemize}
    \item マルチコア計算（共有メモリ, 1ノード）
    \item 大規模分散並列計算（スーパーコンピュータ）
    \item アクセラレータ（Intel Xeon Phi Co-processor, GPU）
    \end{itemize}
  \end{itemize}
\end{frame}

\begin{frame}
  \frametitle{部分固有対計算の方法}
  \input{one_phase_tridiag_evd_en.tex}
  \begin{itemize}
  \item $A$から直接計算するのではなく、以下の3段階に分けて計算
    \begin{enumerate}
    \item 3重対角化 [Wilkinson (1962), etc.]: $A \to T$
    \item 実対称3重対角行列$T$の固有対を計算
    \item 逆変換: $T$の固有ベクトルから$A$の固有ベクトルを計算
    \end{enumerate}
  \end{itemize}
\end{frame}

\begin{frame}
  \frametitle{実対称3重対角行列の固有対計算}
  \begin{exampleblock}{実対称3重対角行列の固有対計算アルゴリズム}
    \begin{itemize}
    \item 全固有対計算アルゴリズム: 
      \begin{itemize}
      \item QR法 [Bowdler-Martin-Reinsch-Wilkinson (1968)]
      \item 分割統治法 [Cuppen (1980) / Gu-Eisenstat (1995)]
      \end{itemize}
    \item 部分固有対計算アルゴリズム: 
      \begin{itemize}
      \item 2分法+逆反復法 [Kahan (1966) / Barth-Martin-Wilkinson (1967)]
        \begin{itemize}
        \item {\color{red}2分法}: 大粒度に並列化でき, 相対精度の意味で高精度な固有値計算法
        \item {\color{blue}逆反復法}: 再直交化計算のため, 計算量が多く並列化も困難
                  \end{itemize}
      \item MRRR法 [Dhillon (1997), etc.]
        \begin{itemize}
        \item 2分法+逆反復法の改良アルゴリズム
        \item 特定の行列では, 計算量が増加してしまう
        \item ロバストに高精度な計算を実現できるわけではない
        \end{itemize}
      \end{itemize}
    \end{itemize}
  \end{exampleblock}
  \begin{alertblock}{大規模並列環境において高速高精度な$T$の部分固有対計算アルゴリズム}
    \begin{itemize}
    \item 2分法+逆反復法なら実現できる?
      $\to$ 逆反復法の改良が必要
      \begin{itemize}
      \item Chapter 3: compact WY 直交化法の導入
      \item Chapter 4: 再直交化付きブロック逆反復法の提案
      \end{itemize}
    \end{itemize}
  \end{alertblock}
\end{frame}

  \begin{block}{Def. 特異対 $(\sigma_i,\ \bm{u}_i,\ \bm{v}_i)$: }
    \begin{itemize}
    \item $m \times n$ real matrix $M$ ($\operatorname{rank}(M)=r,\ r \le \min(m,n)$): 
      \begin{align*}
	\begin{cases}
	  M \bm{v}_i = \sigma_i \bm{u}_i,\\
	  M^\top \bm{u}_i = \sigma_i \bm{v}_i
	\end{cases}
	\quad (i=1,\ \dots,\ r)
      \end{align*}
      \begin{itemize}
      \item $\sigma_i \in \mathbb{R}$: the $i$-th largest singular value for $M$ ($\sigma_1 \ge \dots \ge \sigma_r > 0$)
      \item $\bm{u}_i \in \mathbb{R}^m$: left singular vector corresponding to $\sigma_i$
      \item $\bm{v}_i \in \mathbb{R}^n$: right singular vector corresponding to $\sigma_i$
      \end{itemize}
    \end{itemize}
  \end{block}

      \item $2k$次実対称3重対角行列
      \item 固有対: \textcolor{red}{最大のものから$l$個}の固有値と対応する固有ベクトルのみが必要
      \end{itemize}
    \end{itemize}
  \end{itemize}
  \begin{itemize}
  \item[\checkmark] 上2重対角行列$B_k$の特異対は, 停止条件にも用いられる
  \end{itemize}
